{name:"fert",cost:40,qty:100}, {name:"mulch",cost:35,qty:100}, {name:"tack",cost:25,qty:100} ], premium:[ {name:"bio",cost:25,qty:100}, {name:"humic",cost:20,qty:100} ], addons:[ {name:"lime",cost:15,qty:100}, {name:"sulfur",cost:20,qty:100} ] }, jobs:[], job:{ customer:"", sqft:0, houses:1, package:"standard", addons:{ aeration:false, compost:false, biohum:false, lime:0, sulfur:0, grow:false } } } // ---------- STORAGE ---------- function save(){ localStorage.setItem("app", JSON.stringify(app)) } function load(){ let data = localStorage.getItem("app") if(data) app = JSON.parse(data) } // ---------- LOGIC ---------- function applyPackage(){ let j = app.job if(j.package === "premium"){ j.addons.aeration = true j.addons.compost = true j.addons.biohum = true if(!j.addons.lime) j.addons.lime = 1 if(!j.addons.sulfur) j.addons.sulfur = 1 } } function calculate(){ let j = app.job let sqft = j.sqft let houses = j.houses let factor = sqft / 5000 applyPackage() let used = {} // base if(j.package === "standard"){ used.seed = 0.5 * factor * houses used.fert = 1 * factor * houses used.mulch = 1 * factor * houses used.tack = 0.5 * factor * houses } if(j.package === "premium"){ used.seed = 0.5 * factor * houses used.fert = 1 * factor * houses used.mulch = 1 * factor * houses used.tack = 0.5 * factor * houses used.bio = 0.5 * factor * houses used.humic = 0.5 * factor * houses } // addons if(j.addons.aeration) used.aeration = factor if(j.addons.compost) used.compost = factor if(j.addons.biohum){ used.bio = (used.bio || 0) + 0.5 * factor used.humic = (used.humic || 0) + 0.5 * factor } if(j.addons.lime) used.lime = j.addons.lime if(j.addons.sulfur) used.sulfur = j.addons.sulfur let all = [ ...app.inventory.standard, ...app.inventory.premium, ...app.inventory.addons ] let materialCost = 0 all.forEach(i=>{ if(used[i.name]){ materialCost += used[i.name] * i.cost } }) let labor = (sqft < 5000 ? 200 : 400) * houses let overheadRate = (+el("overhead")?.value || 15)/100 let margin = (+el("margin")?.value || 40)/100 let cost = materialCost + labor let overhead = cost * overheadRate cost += overhead let revenue = cost / (1 - margin) let profit = revenue - cost return {used, materialCost, labor, overhead, cost, revenue, profit} } // ---------- INVENTORY ---------- function findItem(name){ return [...app.inventory.standard, ...app.inventory.premium, ...app.inventory.addons] .find(i => i.name === name) } // ---------- ACTIONS ---------- function createJob(){ let data = calculate() let j = app.job // deduct inventory Object.keys(data.used).forEach(name=>{ let item = findItem(name) if(item) item.qty -= data.used[name] }) app.jobs.push({ name: j.customer || "Customer", revenue: data.revenue, profit: data.profit }) save() render() } // ---------- RENDER ---------- function render(){ let data = calculate() // FINAL CARD el("finalPrice").innerHTML = `
$${Math.round(data.revenue)}
Materials: $${Math.round(data.materialCost)}
Labor: $${data.labor}
Overhead: $${Math.round(data.overhead)}

Cost: $${Math.round(data.cost)}
Profit: $${Math.round(data.profit)}
Margin: ${Math.round((data.profit/data.revenue)*100)}% ` // HISTORY el("history").innerHTML = app.jobs.map(j=>`
${j.name} - $${Math.round(j.revenue)}
`).join("") } // ---------- UI BINDINGS ---------- function bind(){ el("customer").oninput = e => app.job.customer = e.target.value el("sqft").oninput = e => { app.job.sqft = +e.target.value; render() } el("houses").oninput = e => { app.job.houses = +e.target.value; render() } } // ---------- INIT ---------- function init(){ load() bind() render() } init()
